
# Syntactic rules for declaring static text content application
 
# # -> inline comments
# []: -> dict -> contains the content_id of the StyledText
# Indentation rules to define nesting (only 1 level is allowed):

# Style Rules
# * -> italic -> <span style='font-style: italic;'>
# ** -> bold -> <span style='font-weight: bold;'>
# *** -> bold and italic -> <span style='font-weight: bold; font-style: italic;'>
# -- -> line-throug -> <span style='text-decoration: line-throug;'>
# __ -> underline -> <span style='text-decoration: underline;'>
# {value:link} -> link -> <a href='link'> value </a>
# <sub></sub> -> <span style='vertical-align: sub'>
# <sup></sup> -> <span style='vertical-align: sup'>
# $0, $1, ..., $n -> Placeholders that will be dynamically replaced by text in the application.
